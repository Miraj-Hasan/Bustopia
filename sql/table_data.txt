---------------------------- time calculation between two stops in a route which are not consecutive -----------------------------
CREATE OR REPLACE FUNCTION calculate_route_time_simple(
    p_route_id INTEGER,
    p_start_stop VARCHAR,
    p_end_stop VARCHAR
) RETURNS INTERVAL AS $$
DECLARE
    v_stops TEXT[];
    v_start_index INTEGER;
    v_end_index INTEGER;
    v_total_minutes INTEGER := 0;
    v_i INTEGER;
    v_duration INTEGER;
BEGIN
    -- Get the stops array for the specified route
    SELECT stops INTO v_stops FROM Route WHERE route_id = p_route_id;
    
    IF v_stops IS NULL THEN
        RAISE EXCEPTION 'Route % not found', p_route_id;
    END IF;
    
    -- Find the indices of the start and end stops
    v_start_index := array_position(v_stops, p_start_stop);
    v_end_index := array_position(v_stops, p_end_stop);
    
    IF v_start_index IS NULL THEN
        RAISE EXCEPTION 'Start stop "%" not found in route %', p_start_stop, p_route_id;
    END IF;
    
    IF v_end_index IS NULL THEN
        RAISE EXCEPTION 'End stop "%" not found in route %', p_end_stop, p_route_id;
    END IF;
    
    -- Ensure start comes before end
    IF v_start_index > v_end_index THEN
        RAISE EXCEPTION 'Start stop comes after end stop in this route. Use reverse route instead.';
    END IF;
    
    -- Calculate time for each segment
    FOR v_i IN v_start_index..v_end_index-1 LOOP
        SELECT duration INTO v_duration
        FROM Time_mapping 
        WHERE (stop1 = v_stops[v_i] AND stop2 = v_stops[v_i+1])
           OR (stop2 = v_stops[v_i] AND stop1 = v_stops[v_i+1])
        LIMIT 1;
        
        IF v_duration IS NULL THEN
            RAISE EXCEPTION 'Duration not found between % and %', v_stops[v_i], v_stops[v_i+1];
        END IF;
        
        v_total_minutes := v_total_minutes + v_duration;
    END LOOP;
    
    -- Convert minutes to interval
    RETURN (v_total_minutes || ' minutes')::INTERVAL;
END;
$$ LANGUAGE plpgsql;
